// --- NAVIGATION LINKS FOR COLLEGE ADMIN ---
const navLinks = [
  { href: "/college-admin", label: "Dashboard", icon: Home, active: true }, // Dashboard is now always active
  {
    href: "/college-admin/admissions",
    label: "Admissions",
    icon: GraduationCap,
  },
  { href: "/college-admin/students", label: "Students", icon: Users },
  { href: "/college-admin/faculty", label: "Faculty", icon: UserCog },
  { href: "/college-admin/departments", label: "Departments", icon: Building },
  { href: "/college-admin/academics", label: "Academics", icon: BookOpen },
  { href: "/college-admin/timetable", label: "Timetable", icon: CalendarDays },
  { href: "/college-admin/hostel", label: "Hostel", icon: BedDouble },
  { href: "/college-admin/finance", label: "Finance", icon: Wallet },
  { href: "/college-admin/settings", label: "Settings", icon: Settings },
];

export default function CollegeAdminLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  // Theme matching your super-admin style
  const themeConfig = {
    bg: "bg-slate-50",
    card: "bg-white",
    text: "text-slate-900",
    textMuted: "text-slate-600",
    border: "border-slate-200",
    primary: {
      bg: "bg-[#001675]", // Your primary brand color
      text: "text-[#001675]",
    },
  };

  const SidebarNav = () => (
    <nav className="space-y-1 px-4">
      {navLinks.map((link) => (
        <Link
          key={link.label}
          href={link.href}
          className={`group flex items-center gap-3 rounded-xl px-4 py-3 text-sm font-medium transition-all duration-200 ${
            link.active // Use the 'active' property from the navLinks array
              ? `${themeConfig.primary.bg} text-white shadow-md shadow-blue-900/20`
              : `${themeConfig.textMuted} hover:bg-slate-100 hover:text-slate-900`
          }`}
        >
          <link.icon
            className={`h-5 w-5 ${
              link.active ? "text-white" : "text-slate-500"
            }`}
          />
          <span className="flex-1">{link.label}</span>
          {!link.active && (
            <ChevronRight className="h-4 w-4 text-slate-400 opacity-0 group-hover:opacity-100 transition-opacity" />
          )}
        </Link>
      ))}
    </nav>
  );

  return (
    <div className={`min-h-screen ${themeConfig.bg}`}>
      <div className="grid min-h-screen w-full md:grid-cols-[280px_1fr]">
        {/* --- SIDEBAR --- */}
        <div
          className={`hidden border-r ${themeConfig.border} ${themeConfig.card} md:block sticky top-0 h-screen overflow-y-auto`}
        >
          <div className="flex h-full flex-col">
            <div
              className={`flex h-16 items-center border-b ${themeConfig.border} px-6 sticky top-0 ${themeConfig.card} z-10`}
            >
              <Link href="/college-admin" className="flex items-center gap-3">
                <Image
                  src="/logo.png"
                  alt="VidyaLink Logo"
                  width={50}
                  height={50}
                  className="rounded-md"
                />
                <div>
                  <h1 className="font-bold text-xl">
                    <span className="text-[#001675]">Vidya</span>
                    <span className="text-[#444346]">Link</span>
                  </h1>
                  <p className={`text-xs font-medium text-blue-900/70`}>
                    LDRP-ITR Portal
                  </p>
                </div>
              </Link>
            </div>
            <div className="flex-1 overflow-auto py-4">
              <SidebarNav />
            </div>
            <div className={`border-t ${themeConfig.border} p-4 mt-auto`}>
              <Card className={`${themeConfig.bg} ${themeConfig.border}`}>
                <CardContent className="p-3 flex items-center justify-between">
                  <div>
                    <h3 className={`font-semibold text-sm ${themeConfig.text}`}>
                      LDRP-ITR
                    </h3>
                    <p className={`text-xs ${themeConfig.textMuted}`}>
                      Gandhinagar
                    </p>
                  </div>
                  <Image
                    src="/ldrp.png"
                    alt="LDRP-ITR Logo"
                    width={48}
                    height={48}
                    className="rounded-md object-contain"
                  />
                </CardContent>
              </Card>
            </div>
          </div>
        </div>

        {/* --- MAIN CONTENT --- */}
        <div className="flex flex-col">
          <header
            className={`flex h-16 items-center gap-4 border-b ${themeConfig.border} ${themeConfig.card} px-6 justify-between sticky top-0 z-10`}
          >
            {/* Left */}
            <div className="flex items-center">
              <Sheet>
                <SheetTrigger asChild>
                  <Button
                    variant="outline"
                    size="icon"
                    className="shrink-0 md:hidden mr-4"
                  >
                    <Menu className="h-5 w-5" />
                  </Button>
                </SheetTrigger>
                <SheetContent side="left" className="w-72 p-0">
                  {/* Mobile Nav Content */}
                </SheetContent>
              </Sheet>
              <div className="hidden md:block relative">
                <Search className="absolute left-3 top-1/2 h-4 w-4 -translate-y-1/2 text-slate-500" />
                <Input placeholder="Search..." className="pl-10 w-64" />
              </div>
            </div>
            {/* Right */}
            <div className="flex items-center gap-3">
              <DropdownMenu>
                <DropdownMenuTrigger asChild>
                  <Button size="icon" variant="ghost" className="relative">
                    <Bell className="h-5 w-5" />
                    <span className="absolute -top-1 -right-1 h-3 w-3 rounded-full bg-red-500 border-2 border-white"></span>
                  </Button>
                </DropdownMenuTrigger>
                <DropdownMenuContent align="end" className="w-80">
                  {/* Notifications Content */}
                </DropdownMenuContent>
              </DropdownMenu>
              <DropdownMenu>
                <DropdownMenuTrigger asChild>
                  <Button variant="ghost" className="gap-2">
                    <div className="flex items-center gap-2">
                      <div className="hidden md:block text-right">
                        <p className="text-sm font-medium">Ankit Patel</p>
                        <p className="text-xs text-slate-500">College Admin</p>
                      </div>
                      <div
                        className={`h-8 w-8 rounded-full ${themeConfig.primary.bg} flex items-center justify-center`}
                      >
                        <CircleUser className="h-5 w-5 text-white" />
                      </div>
                    </div>
                  </Button>
                </DropdownMenuTrigger>
                <DropdownMenuContent align="end" className="w-56">
                  {/* User Menu Content */}
                </DropdownMenuContent>
              </DropdownMenu>
            </div>
          </header>
          <main className="flex-1 overflow-auto">{children}</main>
        </div>
      </div>
    </div>
  );
}
